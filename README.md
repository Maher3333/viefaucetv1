$ apt install python3

$ apt install pip

$ pip install colorama

$ pip install lxml

$ pip install request

$ pip install bs4

$ pip install platform

$ termux-setup-storage

$ cd /sdcard/downloads

$ git clone https://github.com/tudineroweb/viefaucetv1.git

$ cd viefaucetv1
